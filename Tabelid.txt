
CREATE TABLE looma_omanikud(
	oid INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	nimi VARCHAR(50),
	o_looma_id INT
);

CREATE TABLE loomad(
	lid INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	looma_nimi VARCHAR(50),
	omaniku_id INT,
	looma_liik VARCHAR(50),
	Looma_kirjeldus VARCHAR(255)
);

CREATE TABLE loomaarstid(
	aid INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	arsti_nimi VARCHAR(50),
	alooma_id INT
);

CREATE TABLE protseduurid(
	protseduuri_nimi VARCHAR(50),
	arsti_id INT,
	plooma_id INT
);

CREATE TABLE vastuv6tuajad(
	aeg TIMESTAMP,
	vlooma_id INT
);

CREATE TABLE operatsioonid(
	operatsiooni_nimi VARCHAR(50),
	raskustase VARCHAR(50),
	arsti_id INT,
	olooma_id INT,
	kuup2ev TIMESTAMP
);
